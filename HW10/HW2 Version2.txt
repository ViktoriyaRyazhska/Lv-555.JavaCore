// Програма для шифрурування

package com.company;


public class Main {

    public static StringBuilder encrypt(String s, int n) {
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (Character.isUpperCase(s.charAt(i))) {
                char ch = (char)(((int)s.charAt(i) + n - 65) % 26 + 65);
                result.append(ch);
            }

        }
        return result;
    }
    public static void main(String[] args) {
        String s = "ENIGMA";
        int n = 4;
        System.out.println("Text : " + s);
        System.out.println("Seqrity code :" + n);
        System.out.println("Encrypted text :" + encrypt(s, n));
    }
}
