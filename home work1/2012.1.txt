// приклад використання блоків для виконання декількох
import javax.swing.*;
public class Main {

    public static void main(String[] args) {

	    double i, j, d;

	    i = 10;
	    j = 150;

	    if (i != 0) {

	        d = j / i;
	        JOptionPane.showMessageDialog(null, "j / i рівно " + d);
        } // if (i != 0)
    }
}
