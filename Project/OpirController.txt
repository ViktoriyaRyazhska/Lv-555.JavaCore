package sample;

import java.net.URL;
import java.util.ResourceBundle;
import javafx.event.ActionEvent;
import javafx.fxml.FXML;
import javafx.scene.control.Button;
import javafx.scene.control.TextField;

public class Opir {

    @FXML
    private ResourceBundle resources;

    @FXML
    private URL location;

    @FXML
    private TextField txtamper;

    @FXML
    private TextField txtvolt;

    @FXML
    private Button btnresult;

    @FXML
    private TextField txtopir;

    @FXML
    void Add(ActionEvent event) {
        int amper, volt, opir;
        amper = Integer.parseInt(txtamper.getText());
        volt = Integer.parseInt(txtvolt.getText());
        opir = volt/amper;
        txtopir.setText(String.valueOf(opir));
    }

    @FXML
    void initialize() {

    }
}
