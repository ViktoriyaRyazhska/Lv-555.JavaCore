package com.company;

public class Teacher extends Staff {
    final String TYPE_PERSON = "Teacher";
    Teacher(){
        System.out.println(TYPE_PERSON);
    }
    @Override
    public void print() {
        System.out.println("I am a " + TYPE_PERSON);
    }
    @Override
    public void salary() {
        System.out.println("My salary: 15000");
    }

}
